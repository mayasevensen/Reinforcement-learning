\subsection*{Exercise 3}
\begin{textbf}
    yes, we can define an optimal deterministic policy for this MDP. The optimal policy would be:
    \begin{enumerate}
        \item In state $s_0$, take action \text{prep} to move to state $s_1$.
        \item  In state $s_1$, take action \text{sugar} to move to state $s_2$.
        \item In state $s_2$, take action \text{cook} to move to state $s_6$ and receive a reward of +10.
    \end{enumerate}
    This policy is optimal because it leads to the highest possible reward of +10 by reaching state $s_6$.
    Any deviation from this policy would either lead to a lower reward (e.g., going to state $s_7$ with a reward of -10), a delayed reward (like taking action \text{taste} in $s_2$ leading to state $s_4$), or no reward at all (e.g., waiting in state $s_0$ or taking the restart action in states $s_2$ or $s_3$). Therefore, the optimal deterministic policy is to always choose the actions that lead to state $s_6$.
    This policy is simply the shortest path to the highest reward.
\end{textbf}

\subsection*{Exercise 4}
\begin{textbf}
    Yes, an optimal stochastic policy exists. 
    However, since the environment is fully deterministic and there is a unique optimal trajectory leading to reward +10,
    any randomness that assigns non-zero probability to suboptimal actions (e.g., adding salt) will reduce the expected return.
Thus, the optimal stochastic policy is actually just deterministic, meaning the agent always chooses the best action with probability 1 and never chooses any other actions.
    \[
\pi(a \mid s) =
\begin{cases}
1 & \text{if } a = a^*(s), \\
0 & \text{otherwise}.
\end{cases}
\]

In particular, the agent always chooses actions that lead to the sugary cookies and avoids all actions leading to the salty outcome.
\end{textbf}
\subsection*{Exercise 5}

We define the MDP as
\[
\mathcal{M} = (\mathcal{S}, \mathcal{A}, P, R, \gamma).
\]

\paragraph{States}
To retain the Markov property, we include the latent variable $\sigma \in \{\text{sugar}, \text{salt}\}$:
\[
\mathcal{S} = \{
s_0, s_1, (s_2,\sigma), s_4, s_5, s_6, s_7
\},
\]
where
\[
\begin{aligned}
s_0 &: \text{empty bowl}, \quad
s_1 : \text{full bowl}, \\
(s_2,\sigma) &: \text{enriched bowl with ingredient } \sigma, \\
s_4 &: \text{sugar sampling}, \quad
s_5 : \text{salt sampling}, \\
s_6 &: \text{baked sugary cookies}, \quad
s_7 : \text{baked salty cookies}.
\end{aligned}
\]

\paragraph{Actions}
\[
\begin{aligned}
\mathcal{A}(s_0) &= \{\text{wait}, \text{prepare dough}\}, \\
\mathcal{A}(s_1) &= \{\text{wait}, \text{add ingredient}\}, \\
\mathcal{A}((s_2,\sigma)) &= \{\text{taste}, \text{cook}, \text{restart}\}, \\
\mathcal{A}(s_4) &= \mathcal{A}(s_5) = \{\text{cook}, \text{return}\}, \\
\mathcal{A}(s_6) &= \mathcal{A}(s_7) = \emptyset.
\end{aligned}
\]

\paragraph{Transitions}
\[
\begin{aligned}
&(s_0,\text{wait}) \to s_0, \quad (s_0,\text{prepare dough}) \to s_1, \\
&(s_1,\text{add ingredient}) \to 
\begin{cases}
(s_2,\text{sugar}) & \text{with prob } 0.5, \\
(s_2,\text{salt}) & \text{with prob } 0.5,
\end{cases} \\
&((s_2,\sigma),\text{taste}) \to 
\begin{cases}
s_4 & \text{if } \sigma=\text{sugar}, \\
s_5 & \text{if } \sigma=\text{salt},
\end{cases} \\
&((s_2,\sigma),\text{cook}) \to 
\begin{cases}
s_6 & \text{if } \sigma=\text{sugar}, \\
s_7 & \text{if } \sigma=\text{salt},
\end{cases} \\
&((s_2,\sigma),\text{restart}) \to 
\begin{cases}
(s_2,\text{sugar}) & \text{with prob } 0.5, \\
(s_2,\text{salt}) & \text{with prob } 0.5,
\end{cases} \\
&(s_4,\text{cook}) \to s_6, \quad (s_4,\text{return}) \to (s_2,\text{sugar}), \\
&(s_5,\text{cook}) \to s_7, \quad (s_5,\text{return}) \to (s_2,\text{salt}).
\end{aligned}
\]

\paragraph{Reward function}
\[
R(s,a,s') =
\begin{cases}
+10 & \text{if } s' = s_6, \\
-10 & \text{if } s' = s_7, \\
0 & \text{otherwise}.
\end{cases}
\]
